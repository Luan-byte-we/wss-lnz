import websocket, json, threading

def on_open(ws):
    print("✅ Conectado à WSS!")
    ws.send(json.dumps({
        "bot":     "python",
        "dados":   "meu brainrot",
        "extra":   {"qualquer": "coisa"}
    }))

def on_message(ws, msg):
    data = json.loads(msg)
    print(f"Recebido: {data}")

def on_error(ws, err):
    print(f"Erro: {err}")

def on_close(ws, code, msg):
    print("Desconectado")

ws = websocket.WebSocketApp(
    "wss://ws-LNZ.online/on",
    on_open=on_open,
    on_message=on_message,
    on_error=on_error,
    on_close=on_close
)
ws.run_forever()

const { Client, GatewayIntentBits } = require('discord.js')
const WebSocket = require('ws')

const bot = new Client({ intents: [GatewayIntentBits.Guilds, GatewayIntentBits.GuildMessages, GatewayIntentBits.MessageContent] })

let wss = null

function connectWss() {
    wss = new WebSocket('wss://ws-LNZ.online/on')
    wss.on('open', () => console.log('✅ WSS online!'))
    wss.on('close', () => setTimeout(connectWss, 5000)) // reconecta
}

bot.on('ready', () => {
    console.log(`Bot: ${bot.user.tag}`)
    connectWss()
})

bot.on('messageCreate', msg => {
    if (msg.author.bot) return
    if (msg.content.startsWith('!enviar ')) {
        const texto = msg.content.slice(8)
        wss?.send(JSON.stringify({
            bot:    'discord',
            autor:  msg.author.tag,
            guild:  msg.guild?.name,
            dados:  texto,
        }))
        msg.reply('📡 Enviado para a WSS!')
    }
})

// go get github.com/gorilla/websocket
package main

import (
    "encoding/json"
    "fmt"
    "github.com/gorilla/websocket"
    "net/url"
)

func main() {
    u := url.URL{Scheme: "wss", Host: "ws-LNZ.online", Path: "/on"}
    c, _, err := websocket.DefaultDialer.Dial(u.String(), nil)
    if err != nil {
        panic(err)
    }
    defer c.Close()

    payload, _ := json.Marshal(map[string]interface{}{
        "bot":   "golang",
        "dados": "brainrot do Go",
    })

    c.WriteMessage(websocket.TextMessage, payload)

    _, msg, _ := c.ReadMessage()
    fmt.Printf("Resposta: %s\n", msg)
}
