💻 Exemplos de Uso

Endpoint WSS: ws://localhost:3000/ws (local) 
import asyncio
import websockets
import json

async def send_data():
    uri = "ws://localhost:3000/ws"  # ou wss://seu-dominio.com/ws
    
    async with websockets.connect(uri) as websocket:
        # Enviar JSON
        await websocket.send(json.dumps({"tipo": "brainrot", "conteudo": "teste"}))
        
        # Enviar texto puro
        await websocket.send("Texto simples")
        
        # Enviar número
        await websocket.send("12345")
        
        # Enviar HTML
        await websocket.send("<h1>Teste</h1>")

# Executar
asyncio.run(send_data())

const WebSocket = require('ws');

const ws = new WebSocket('ws://localhost:3000/ws');
// ou wss://seu-dominio.com/ws

ws.on('open', () => {
    // Enviar JSON
    ws.send(JSON.stringify({tipo: "brainrot", conteudo: "teste"}));
    
    // Enviar texto
    ws.send("Texto simples");
    
    // Enviar número
    ws.send("12345");
    
    // Enviar HTML
    ws.send("<h1>Teste</h1>");
});

ws.on('message', (msg) => {
    console.log('Recebido:', msg);
});

ws.on('close', () => {
    console.log('Desconectado');
});
