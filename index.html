import asyncio
import websockets
import json

async def send_data():
    uri = "wss://wss-lnz-production.up.railway.app/ws"
    async with websockets.connect(uri) as websocket:
        # Enviar JSON
        await websocket.send(json.dumps({"tipo": "brainrot", "conteudo": "teste"}))
        
        # Enviar texto puro
        await websocket.send("Isso é um texto simples")
        
        # Enviar número
        await websocket.send("12345")
        
        # Receber resposta
        response = await websocket.recv()
        print(response)

asyncio.run(send_data())
const WebSocket = require('ws');
