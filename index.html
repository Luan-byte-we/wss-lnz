import websocket, json
ws = websocket.WebSocket()

ws.close()

# Node.js
const WebSocket = require('ws');
const ws = new WebSocket('wss://ws-LNZ.online/on');
ws.on('open', () => ws.send(JSON.stringify({bot: 'node', msg: 'teste'})));
ws.on('message', d => console.log(d.toString()));
