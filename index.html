💻 Exemplos de Código para Conectar seus Bots

🐍 Python

import asyncio import websockets import json async def enviar_dados(): uri = "wss://wss-lnz-production.up.railway.app/ws" async with websockets.connect(uri) as websocket: # Envia dados dados = { "tipo": "brainrot", "conteudo": "Meu primeiro brainrot!", "autor": "Python Bot" } await websocket.send(json.dumps(dados)) # Recebe confirmação resposta = await websocket.recv() print(f"Resposta: {resposta}") asyncio.run(enviar_dados())

🟢 Node.js

const WebSocket = require('ws'); const ws = new WebSocket('wss://wss-lnz-production.up.railway.app/ws'); ws.on('open', () => { console.log('Conectado!'); // Envia dados const dados = { tipo: "brainrot", conteudo: "Meu primeiro brainrot!", autor: "Node.js Bot" }; ws.send(JSON.stringify(dados)); }); ws.on('message', (data) => { console.log('Resposta:', JSON.parse(data)); });

🌐 JavaScript (Browser)

const ws = new WebSocket('wss://wss-lnz-production.up.railway.app/ws'); ws.onopen = () => { console.log('Conectado!'); ws.send(JSON.stringify({ tipo: "brainrot", conteudo: "Teste do navegador!", autor: "Browser" })); }; ws.onmessage = (e) => { console.log('Resposta:', JSON.parse(e.data)); };
